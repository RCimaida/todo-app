from flask import Flask, render_template, request, redirect
import json
import os

app = Flask(__name__)

def load_todos():
    if not os.path.exists("todos.json"):
        return []
    with open("todos.json", "r") as f:
        return json.load(f)

def save_todos(todos):
    with open("todos.json", "w") as f:
        json.dump(todos, f, ensure_ascii=False)

@app.route("/")
def index():
    todos = load_todos()
    return render_template("index.html", todos=todos)

@app.route("/add", methods=["POST"])
def add():
    todos = load_todos()
    title = request.form.get("title")
    deadline = request.form.get("deadline")  # 締め切り日を受け取る
    if title:
        todo = {
            "id": len(todos) + 1,
            "title": title,
            "deadline": deadline,  # 締め切り日を保存
            "done": False
        }
        todos.append(todo)
        save_todos(todos)
    return redirect("/")

@app.route("/complete/<int:todo_id>")
def complete(todo_id):
    todos = load_todos()
    for todo in todos:
        if todo["id"] == todo_id:
            todo["done"] = True
    save_todos(todos)
    return redirect("/")

@app.route("/delete/<int:todo_id>")
def delete(todo_id):
    todos = load_todos()
    todos = [t for t in todos if t["id"] != todo_id]
    save_todos(todos)
    return redirect("/")

if __name__ == "__main__":
    app.run(debug=True)
